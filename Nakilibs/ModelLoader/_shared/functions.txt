local NL = {}

function NL.autoFilterDropboxLinks(url)
    if url:sub(9,23) == "www.dropbox.com" then
        urlNew=string.replace(url,"www.dropbox.com","dl.dropboxusercontent.com")
        if urlNew:sub(-4,-1) == "dl=0" then
            urlNew=string.replace(urlNew,"dl=0","dl=1")
            --print("detected unchanged dropbox link, replacing dropbox with dropboxusercontent and dl=0 to dl=1")
        else
            --print("detected unchanged dropbox link, replacing dropbox with dropboxusercontent")
        end
    else
        urlNew=url
    end
    return urlNew
end

function NL.permsCheck(permsTable,networkToServer)
    local failedPerms = {}
    
    local allPermsOK = true
    for k, v in pairs(permsTable) do
        local permOK = hasPermission(v, owner())
        if allPermsOK == false then
            allPermsOK = false
            failedPerms[#failedPerms+1] = v
        end
    end
    if not allPermsOK then
        net.start("NL:permsCheckFail")
        net.writeTable(failedPerms)
        net.send()
        
        error("Missing permissions: "..table.concat(failedPerms, ","), 0)
    end
    
    return allPermsOK
end

return NL
