--@author Nakkitsunami
--@client

local waila = {}

waila.color_white = Color(255)
waila.color_light_gray = Color(180,180,180)

waila.icons = {
    ["_NOICON"] = render.createMaterial("gui/noicon.png"),
}
waila.icons["heart"] = waila.icons["_NOICON"]
waila.icons["heart_empty"] = waila.icons["_NOICON"]

local path = file.existsTemp("waila_heart.png")
if path then
    waila.icons["heart"] = material.createFromImage(path, "nocull noclamp")
else
    http.get("https://dl.dropboxusercontent.com/scl/fi/caol9o6m5h4ejeep3c7u1/heart.png?rlkey=a1q3niou4vnxvgoottim4yzqo&dl=1", function(getData)
        local path = file.writeTemp("waila_heart.png", getData)
        waila.icons["heart"] = material.createFromImage(path, "nocull noclamp")
    end)
end

path = file.existsTemp("waila_heart_empty.png")
if path then
    waila.icons["heart_empty"] = material.createFromImage(path, "nocull noclamp")
else
    http.get("https://dl.dropboxusercontent.com/scl/fi/pq1mjlfl56omdl17b3p1m/heart_empty.png?rlkey=kv00nm79ofid1ubfkum43w8hv&dl=1", function(getData)
        local path = file.writeTemp("waila_heart_empty.png", getData)
        waila.icons["heart_empty"] = material.createFromImage(path, "nocull noclamp")
    end)
end

function waila.name(ent)
    local text = ent:getName()
    render.setColor(waila.color_white)
    render.drawSimpleText(0, 0, text, TEXT_ALIGN.LEFT, TEXT_ALIGN.TOP)
    return render.getTextSize(text)
end

function waila.health(ent)
    local hp = ent:getHealth()
    local max_hp = ent:getMaxHealth()
    local hp_ratio = hp/max_hp
    local _, get_height = render.getTextSize("height")
    
    render.setColor(waila.color_white)
    render.setMaterial(waila.icons["heart"])
    for i = 1, 10 do
        if math.ceil(hp_ratio*10,0) < i then
            render.setMaterial(waila.icons["heart_empty"])
        end
        render.drawTexturedRectFast(25*(i-1), 0, 25, 25)
    end
    return 25*10, 25
end

function waila.armor(ent)
    local text = tostring(ent:getArmor()).."/"..tostring(ent:getMaxArmor())
    render.setColor(waila.color_light_gray)
    render.drawSimpleText(0, 0, text, TEXT_ALIGN.LEFT, TEXT_ALIGN.TOP)
    return render.getTextSize(text)
end

function waila.rank(ent)
    local text = ent:getTeamName()
    render.setColor(team.getColor(ent:getTeam()))
    render.drawSimpleText(0, 0, text, TEXT_ALIGN.LEFT, TEXT_ALIGN.TOP)
    return render.getTextSize(text)
end

function waila.class(ent)
    local text = ent:getClass()
    render.drawSimpleText(0, 0, text, TEXT_ALIGN.LEFT, TEXT_ALIGN.TOP)
    return render.getTextSize(text)
end

function waila.owner(ent)
    local owner = ent:getOwner()
    local text = (not owner or owner == game.getWorld()) and "WORLD" or owner:getName()
    render.drawSimpleText(0, 0, text, TEXT_ALIGN.LEFT, TEXT_ALIGN.TOP)
    return render.getTextSize(text)
end

function waila.model(ent)
    local text = ent:getModel()
    render.drawSimpleText(0, 0, text, TEXT_ALIGN.LEFT, TEXT_ALIGN.TOP)
    return render.getTextSize(text)
end

function waila.material(ent)
    local mat = ent:getMaterial()
    local text = mat == "" and "NONE" or mat
    render.drawSimpleText(0, 0, text, TEXT_ALIGN.LEFT, TEXT_ALIGN.TOP)
    return render.getTextSize(text)
end

function waila.color(ent)
    local color = ent:getColor()
    local text = tostring(color)
    render.setColor(color:setA(255))
    render.drawSimpleText(0, 0, text, TEXT_ALIGN.LEFT, TEXT_ALIGN.TOP)
    return render.getTextSize(text)
end

function waila.weapon(ent)
    local weapon = ent:getActiveWeapon()
    local text = weapon == nil and "Unarmed" or weapon:getPrintName()
    render.drawSimpleText(0, 0, text, TEXT_ALIGN.LEFT, TEXT_ALIGN.TOP)
    return render.getTextSize(text)
end


function waila.pushMask(mask, invert)
    render.setStencilWriteMask(0xFF)
    render.setStencilTestMask(0xFF)
    render.setStencilReferenceValue(0)
    render.setStencilCompareFunction(8)
    render.setStencilPassOperation(1)
    render.setStencilFailOperation(1)
    render.setStencilZFailOperation(1)
    render.clearStencil()
    
    render.setStencilEnable(true)
    
    render.setStencilReferenceValue(1)
    render.setStencilCompareFunction(1)
    render.setStencilFailOperation(3)
    
    mask()
    
    render.setStencilCompareFunction(invert and STENCIL.EQUAL or STENCIL.NOTEQUAL)
    render.setStencilFailOperation(1)
end

function waila.popMask()
    render.setStencilEnable(false)
end

return waila
